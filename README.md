'  11    '' 8    8 ''     ''4   4   '' 9    9 '
' 1 1    ''  8888  '' []  ''4444444 ''  99999 '
'   1    ''  8888  ''     ''    4   ''      9 '
'   1    '' 8    8 '' []  ''    4   ''      9 '
' 11111  ''  8888  ''     ''    4   ''  9999  '
